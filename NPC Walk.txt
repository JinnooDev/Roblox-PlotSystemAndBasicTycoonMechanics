local function  SpawnNPC()

	NPC.HumanoidRootPart.CFrame = StartLocation.CFrame + Vector3.new(0, 5, 0)
	
	for _, v in ipairs(Dummy:GetDescendants()) do
		print(v.Name)
		print(TriggerPart.Parent)
	end
	
	local TriggerPart = NPC:FindFirstChild("TriggerPart", true)

	TriggerPart.Transparency = 1
	TriggerPart.Color = Color3.new(1,0,0)
	TriggerPart.Size = Vector3.new(20,20,20)
	
	wait(1)
	MoveNPC()
	
	
	
	
	
	
	
	
end


while true do
